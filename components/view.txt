import React, { useEffect, useState } from 'react';
import { View, Text, FlatList } from 'react-native';
import { getScannedData } from './utils/storage';

export default function HistoryScreen() {
  const [data, setData] = useState([]);

  useEffect(() => {
    const load = async () => {
      const saved = await getScannedData();
      setData(saved.reverse());
    };
    load();
  }, []);

  return (
    <View style={{ padding: 20 }}>
      <Text style={{ fontWeight: 'bold', fontSize: 18 }}>Scanned QR History:</Text>
      <FlatList
        data={data}
        keyExtractor={(item) => item.id.toString()}
        renderItem={({ item }) => (
          <View style={{ marginVertical: 5 }}>
            <Text>ğŸ“¦ {item.data}</Text>
            <Text style={{ fontSize: 12, color: 'gray' }}>{item.timestamp}</Text>
          </View>
        )}
      />
    </View>
  );
}
